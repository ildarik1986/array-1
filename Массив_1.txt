#include <stdio.h>
#define N 10
main() {
    int a[N];
    int i, min1, min2, buff, sum;
    srand(time(NULL));
    for (i=0; i<N; i++) {
        a[i] = rand() % 100;
        printf("%3d", a[i]);
    }
    printf("\n");

    if (a[0] < a[1]) {
        min1 = 0;
        min2 = 1;
    } else {
        min1 = 1;
        min2 = 0;
    }

    for (i=2; i<N; i++) {
        if (a[i] < a[min1]) {
            buff = min1;
            min1 = i;
            if (a[buff] < a[min2]) min2 = buff;
        } else {
            if (a[i] < a[min2]) min2 = i;
        }
    }

    printf("№%2d:%3d\n",min1+1,a[min1]);
    printf("№%2d:%3d\n",min2+1,a[min2]);
}
	sum = 0;
    sum += (min1+min2);
    printf("%4d\n", sum);
